					ADMIN HISTORY REQUEST VIEW

from django.db.models.functions import TruncMonth
from django.db.models import Count, Sum
from django.shortcuts import render
from .models import BloodRequest  # بررسی کنید مدل شما همین نام را داشته باشد
from donor.models import BloodDonate  # بررسی کنید مدل شما همین نام را داشته باشد

def admin_request_history_view(request):
    # دریافت داده‌های ماهانه برای دریافت خون (درخواست توسط بیمار)
    received_blood = BloodRequest.objects.filter(request_by_patient__isnull=False).annotate(
        month=TruncMonth('date')
    ).values('month').annotate(
        total_requests=Count('id'),
        total_units=Sum('unit')
    ).order_by('-month')

    # دریافت داده‌های ماهانه برای اهدا خون (تغییر `request_by_donor` به `donor`)
    donated_blood = BloodDonate.objects.filter(donor__isnull=False).annotate(
        month=TruncMonth('date')
    ).values('month').annotate(
        total_donations=Count('id'),
        total_donated_units=Sum('unit')
    ).order_by('-month')

    return render(request, 'blood/admin_request_history.html', {
        'received_blood': received_blood,
        'donated_blood': donated_blood
    })
